\subsection{Algoritmo}




%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}{Título del tema}{Título de la sección}
\vskip 0.3cm % Para subir (-) o bajar el texto
\textbf{Título de la subsección (o del concepto en su caso)}
	\begin{algoritmo}[H]	
	\algsetup{linenosize=\tiny}		
	\caption{\scriptsize{Seleccion Clonal}}
	\label{alg:SeleccionClonal}
	\scriptsize
		\begin{algorithmic}[1]
			\REQUIRE $T\geq1$ \COMMENT{Número de iteraciones}
			\REQUIRE $N\geq1$ \COMMENT{Número de células en la población}
			\REQUIRE $M\geq1$ \COMMENT{Número de células seleccionadas para clonarse}			
			\REQUIRE $\beta>0$ \COMMENT{Factor multiplicativo para el número de clones}
			\REQUIRE $R\geq1$ \COMMENT{Número de células aleatorias}			
			\REQUIRE $\rho>0$ \COMMENT{Tasa de decaimiento de la mutación} 
			\ENSURE $S$ \COMMENT{Mejor solución encontrada}
			\STATE $P_0\leftarrow$ RandomPopulation($N$) \COMMENT{Crea una población de células aleatoria}		
			\STATE $t\leftarrow0$
			\WHILE {$t<T$}
				\STATE $B_t\leftarrow$ Best($P_t$, $M$) \COMMENT{Selecciona las mejores $M$ células de $P_t$}
				\STATE $C_t\leftarrow$ Clones($B_t$, $\beta$, $N$) \COMMENT{Genera clones}
				\STATE $O_t\leftarrow$ Maturate($C_t$, $\rho$) \COMMENT{Maduración de los clones vía hipermutación}
				\STATE $Q_t\leftarrow$ RandomPopulation($R$) \COMMENT{Crea una población de células aleatoria}				
				\STATE $P_{t+1}\leftarrow$ Best($P_t+O_t+Q_t$, $N$) \COMMENT{Selecciona las $N$ mejores células de $P_t+O_t+Q_t$}
				\STATE $t\leftarrow t+1$
			\ENDWHILE
			\STATE $S\leftarrow$ Best($P_t$)  \COMMENT{Devuelve la mejor solución}
			\RETURN $S$
		\end{algorithmic}
	\end{algoritmo}
\end{frame}
